             xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"

  <process id="wrc_process" name="Warranty-Replacement v1.3" isExecutable="false">

    <startEvent id="wrc_start" name="Customer submits an online warranty claim">
      <outgoing>wrc_flow_start_validate</outgoing>
    </startEvent>

    <task id="wrc_validate" name="validates the claim">
      <incoming>wrc_flow_start_validate</incoming>
      <incoming>wrc_flow_request_revalidate</incoming>
      <outgoing>wrc_flow_validate_completeness</outgoing>
    </task>

    <exclusiveGateway id="wrc_gwCompleteness" name="Claim complete?" gatewayDirection="Diverging">
      <incoming>wrc_flow_validate_completeness</incoming>
      <outgoing>wrc_flow_incomplete_request</outgoing>
      <outgoing>wrc_flow_complete_check</outgoing>
    </exclusiveGateway>

    <task id="wrc_requestInfo" name="requests missing information">
      <incoming>wrc_flow_incomplete_request</incoming>
      <outgoing>wrc_flow_request_revalidate</outgoing>
    </task>

    <task id="wrc_checkEligibility" name="checks warranty eligibility">
      <incoming>wrc_flow_complete_check</incoming>
      <outgoing>wrc_flow_check_warranty</outgoing>
    </task>

    <exclusiveGateway id="wrc_gwWarranty" name="Within warranty?" gatewayDirection="Diverging">
      <incoming>wrc_flow_check_warranty</incoming>
      <outgoing>wrc_flow_out_reject</outgoing>
      <outgoing>wrc_flow_in_split</outgoing>
    </exclusiveGateway>

    <task id="wrc_rejectionNotice" name="send rejection notice">
      <incoming>wrc_flow_out_reject</incoming>
      <outgoing>wrc_flow_reject_end</outgoing>
    </task>

    <endEvent id="wrc_endRejected" name="rejection">
      <incoming>wrc_flow_reject_end</incoming>
    </endEvent>

    <parallelGateway id="wrc_gwParallelSplit" gatewayDirection="Diverging">
      <incoming>wrc_flow_in_split</incoming>
      <outgoing>wrc_flow_split_create</outgoing>
      <outgoing>wrc_flow_split_schedule</outgoing>
    </parallelGateway>

    <task id="wrc_createShipment" name="Create shipment order">
      <incoming>wrc_flow_split_create</incoming>
      <outgoing>wrc_flow_create_join</outgoing>
    </task>

    <task id="wrc_schedulePickup" name="Schedule courier pickup">
      <incoming>wrc_flow_split_schedule</incoming>
      <outgoing>wrc_flow_schedule_join</outgoing>
    </task>

    <parallelGateway id="wrc_gwParallelJoin" gatewayDirection="Converging">
      <incoming>wrc_flow_create_join</incoming>
      <incoming>wrc_flow_schedule_join</incoming>
      <outgoing>wrc_flow_join_update</outgoing>
    </parallelGateway>

    <task id="wrc_updateInventory" name="update inventory records">
      <incoming>wrc_flow_join_update</incoming>
      <outgoing>wrc_flow_update_close</outgoing>
    </task>

    <task id="wrc_closeClaim" name="close the claim">
      <incoming>wrc_flow_update_close</incoming>
      <outgoing>wrc_flow_close_end</outgoing>
    </task>

    <endEvent id="wrc_endSuccess" name="successful completion">
      <incoming>wrc_flow_close_end</incoming>
    </endEvent>

    <sequenceFlow id="wrc_flow_start_validate"
                  sourceRef="wrc_start"
                  targetRef="wrc_validate"/>

    <sequenceFlow id="wrc_flow_validate_completeness"
                  sourceRef="wrc_validate"
                  targetRef="wrc_gwCompleteness"/>

    <sequenceFlow id="wrc_flow_incomplete_request"
                  name="incomplete"
                  sourceRef="wrc_gwCompleteness"
                  targetRef="wrc_requestInfo">
      <conditionExpression xsi:type="bpmn:tFormalExpression">claim is incomplete</conditionExpression>
    </sequenceFlow>

    <sequenceFlow id="wrc_flow_complete_check"
                  name="complete"
                  sourceRef="wrc_gwCompleteness"
                  targetRef="wrc_checkEligibility">
      <conditionExpression xsi:type="bpmn:tFormalExpression">claim is complete</conditionExpression>
    </sequenceFlow>

    <sequenceFlow id="wrc_flow_request_revalidate"
                  name="information received"
                  sourceRef="wrc_requestInfo"
                  targetRef="wrc_validate"/>

    <sequenceFlow id="wrc_flow_check_warranty"
                  sourceRef="wrc_checkEligibility"
                  targetRef="wrc_gwWarranty"/>

    <sequenceFlow id="wrc_flow_out_reject"
                  name="out of warranty"
                  sourceRef="wrc_gwWarranty"
                  targetRef="wrc_rejectionNotice">
      <conditionExpression xsi:type="bpmn:tFormalExpression">product is out of warranty</conditionExpression>
    </sequenceFlow>

    <sequenceFlow id="wrc_flow_reject_end"
                  sourceRef="wrc_rejectionNotice"
                  targetRef="wrc_endRejected"/>

    <sequenceFlow id="wrc_flow_in_split"
                  name="within warranty"
                  sourceRef="wrc_gwWarranty"
                  targetRef="wrc_gwParallelSplit">
      <conditionExpression xsi:type="bpmn:tFormalExpression">product is within warranty</conditionExpression>
    </sequenceFlow>

    <sequenceFlow id="wrc_flow_split_create"
                  sourceRef="wrc_gwParallelSplit"
                  targetRef="wrc_createShipment"/>

    <sequenceFlow id="wrc_flow_split_schedule"
                  sourceRef="wrc_gwParallelSplit"
                  targetRef="wrc_schedulePickup"/>

    <sequenceFlow id="wrc_flow_create_join"
                  sourceRef="wrc_createShipment"
                  targetRef="wrc_gwParallelJoin"/>

    <sequenceFlow id="wrc_flow_schedule_join"
                  sourceRef="wrc_schedulePickup"
                  targetRef="wrc_gwParallelJoin"/>

    <sequenceFlow id="wrc_flow_join_update"
                  sourceRef="wrc_gwParallelJoin"
                  targetRef="wrc_updateInventory"/>

    <sequenceFlow id="wrc_flow_update_close"
                  sourceRef="wrc_updateInventory"
                  targetRef="wrc_closeClaim"/>

    <sequenceFlow id="wrc_flow_close_end"
                  sourceRef="wrc_closeClaim"
                  targetRef="wrc_endSuccess"/>

  </process>

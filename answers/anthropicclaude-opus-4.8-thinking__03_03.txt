    <bpmn:exclusiveGateway id="wrc_gw_completeness" name="Claim complete?">

      <bpmn:outgoing>wrc_flow_incomplete</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_complete</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_task_request_missing_info" name="requests missing information">

    </bpmn:task>

    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_task_send_rejection" name="send rejection notice">

      <bpmn:outgoing>wrc_flow_rejection_to_end</bpmn:outgoing>
    </bpmn:task>

    </bpmn:endEvent>

    </bpmn:parallelGateway>

    <bpmn:task id="wrc_task_update_inventory" name="update inventory records">

      <bpmn:outgoing>wrc_flow_update_to_close</bpmn:outgoing>
    </bpmn:task>

      <bpmn:incoming>wrc_flow_update_to_close</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_close_to_end</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_end_completed" name="Claim closed">
      <bpmn:incoming>wrc_flow_close_to_end</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:sequenceFlow id="wrc_flow_rejection_to_end" sourceRef="wrc_task_send_rejection" targetRef="wrc_end_rejected"/>
    <bpmn:sequenceFlow id="wrc_flow_within_warranty" name="Within warranty" sourceRef="wrc_gw_warranty" targetRef="wrc_gw_parallel_split"/>
    <bpmn:sequenceFlow id="wrc_flow_split_to_shipment" sourceRef="wrc_gw_parallel_split" targetRef="wrc_task_create_shipment_order"/>
    <bpmn:sequenceFlow id="wrc_flow_split_to_courier" sourceRef="wrc_gw_parallel_split" targetRef="wrc_task_schedule_courier_pickup"/>

    <bpmn:sequenceFlow id="wrc_flow_courier_to_join" sourceRef="wrc_task_schedule_courier_pickup" targetRef="wrc_gw_parallel_join"/>
    <bpmn:sequenceFlow id="wrc_flow_join_to_update" sourceRef="wrc_gw_parallel_join" targetRef="wrc_task_update_inventory"/>
    <bpmn:sequenceFlow id="wrc_flow_update_to_close" sourceRef="wrc_task_update_inventory" targetRef="wrc_task_close_claim"/>
    <bpmn:sequenceFlow id="wrc_flow_close_to_end" sourceRef="wrc_task_close_claim" targetRef="wrc_end_completed"/>

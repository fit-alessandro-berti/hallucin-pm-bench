    <bpmn:startEvent id="wrc_start_online_claim" name="online warranty claim">
      <bpmn:outgoing>wrc_flow_start_validate</bpmn:outgoing>
      <bpmn:messageEventDefinition
          id="wrc_message_event_online_claim"
          messageRef="wrc_message_online_claim"/>
    </bpmn:startEvent>

    <bpmn:task id="wrc_task_validate_claim" name="validates the claim">
      <bpmn:incoming>wrc_flow_start_validate</bpmn:incoming>
      <bpmn:incoming>wrc_flow_wait_validate</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_validate_completeness</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway
        id="wrc_gateway_completeness"
        name="Claim complete?"
        gatewayDirection="Diverging">
      <bpmn:incoming>wrc_flow_validate_completeness</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_incomplete_request</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_complete_check</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_task_request_missing_information" name="requests missing information">
      <bpmn:incoming>wrc_flow_incomplete_request</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_request_wait</bpmn:outgoing>
    </bpmn:task>

    <bpmn:intermediateCatchEvent
        id="wrc_event_missing_information_received"
        name="missing information received">
      <bpmn:incoming>wrc_flow_request_wait</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_wait_validate</bpmn:outgoing>
      <bpmn:messageEventDefinition
          id="wrc_message_event_missing_information"
          messageRef="wrc_message_missing_information"/>
    </bpmn:intermediateCatchEvent>

    <bpmn:task id="wrc_task_check_warranty" name="checks warranty eligibility">
      <bpmn:incoming>wrc_flow_complete_check</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_check_warranty_decision</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway
        id="wrc_gateway_warranty_decision"
        name="Within warranty?"
        gatewayDirection="Diverging">
      <bpmn:incoming>wrc_flow_check_warranty_decision</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_out_of_warranty</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_within_warranty</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_task_send_rejection" name="send rejection notice">
      <bpmn:incoming>wrc_flow_out_of_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_rejection_end</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_end_rejection" name="Rejection">
      <bpmn:incoming>wrc_flow_rejection_end</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:parallelGateway
        id="wrc_gateway_parallel_split"
        gatewayDirection="Diverging">
      <bpmn:incoming>wrc_flow_within_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_split_create_shipment</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_split_schedule_pickup</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_task_create_shipment" name="Create shipment order">
      <bpmn:incoming>wrc_flow_split_create_shipment</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_create_shipment_join</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_task_schedule_pickup" name="Schedule courier pickup">
      <bpmn:incoming>wrc_flow_split_schedule_pickup</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_schedule_pickup_join</bpmn:outgoing>
    </bpmn:task>

    <bpmn:parallelGateway
        id="wrc_gateway_parallel_join"
        gatewayDirection="Converging">
      <bpmn:incoming>wrc_flow_create_shipment_join</bpmn:incoming>
      <bpmn:incoming>wrc_flow_schedule_pickup_join</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_join_update_inventory</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_task_update_inventory" name="update inventory records">
      <bpmn:incoming>wrc_flow_join_update_inventory</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_update_inventory_close</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_task_close_claim" name="close the claim">
      <bpmn:incoming>wrc_flow_update_inventory_close</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_close_success</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_end_success" name="Successful completion">
      <bpmn:incoming>wrc_flow_close_success</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:sequenceFlow
        id="wrc_flow_start_validate"

    <bpmn:sequenceFlow
        id="wrc_flow_validate_completeness"

        id="wrc_flow_incomplete_request"

    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow
        id="wrc_flow_request_wait"

    <bpmn:sequenceFlow
        id="wrc_flow_wait_validate"
        sourceRef="wrc_event_missing_information_received"
        targetRef="wrc_task_validate_claim"/>

    <bpmn:sequenceFlow
        id="wrc_flow_complete_check"
        name="Complete"
        sourceRef="wrc_gateway_completeness"
        targetRef="wrc_task_check_warranty">
      <bpmn:conditionExpression
          id="wrc_condition_complete"
          xsi:type="bpmn:tFormalExpression"><![CDATA[claimComplete = true]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow
        id="wrc_flow_check_warranty_decision"
        sourceRef="wrc_task_check_warranty"
        targetRef="wrc_gateway_warranty_decision"/>

    <bpmn:sequenceFlow
        id="wrc_flow_out_of_warranty"
        name="Out of warranty"
        sourceRef="wrc_gateway_warranty_decision"
        targetRef="wrc_task_send_rejection">
      <bpmn:conditionExpression
          id="wrc_condition_out_of_warranty"
          xsi:type="bpmn:tFormalExpression"><![CDATA[withinWarranty = false]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow
        id="wrc_flow_rejection_end"
        sourceRef="wrc_task_send_rejection"
        targetRef="wrc_end_rejection"/>

    <bpmn:sequenceFlow
        id="wrc_flow_within_warranty"
        name="Within warranty"
        sourceRef="wrc_gateway_warranty_decision"
        targetRef="wrc_gateway_parallel_split">
      <bpmn:conditionExpression
          id="wrc_condition_within_warranty"
          xsi:type="bpmn:tFormalExpression"><![CDATA[withinWarranty = true]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow
        id="wrc_flow_split_create_shipment"
        sourceRef="wrc_gateway_parallel_split"
        targetRef="wrc_task_create_shipment"/>

    <bpmn:sequenceFlow
        id="wrc_flow_split_schedule_pickup"
        sourceRef="wrc_gateway_parallel_split"
        targetRef="wrc_task_schedule_pickup"/>

    <bpmn:sequenceFlow
        id="wrc_flow_create_shipment_join"
        sourceRef="wrc_task_create_shipment"
        targetRef="wrc_gateway_parallel_join"/>

    <bpmn:sequenceFlow
        id="wrc_flow_schedule_pickup_join"
        sourceRef="wrc_task_schedule_pickup"
        targetRef="wrc_gateway_parallel_join"/>

    <bpmn:sequenceFlow
        id="wrc_flow_join_update_inventory"
        sourceRef="wrc_gateway_parallel_join"
        targetRef="wrc_task_update_inventory"/>

    <bpmn:sequenceFlow
        id="wrc_flow_update_inventory_close"
        sourceRef="wrc_task_update_inventory"
        targetRef="wrc_task_close_claim"/>

    <bpmn:sequenceFlow
        id="wrc_flow_close_success"
        sourceRef="wrc_task_close_claim"
        targetRef="wrc_end_success"/>

  </bpmn:process>

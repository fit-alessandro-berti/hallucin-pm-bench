    <task id="wrc_task_update_inventory" name="update inventory records" />

    <sequenceFlow id="wrc_sf_start_validate" sourceRef="wrc_start" targetRef="wrc_task_validate" />

    <sequenceFlow id="wrc_sf_schedule_join" sourceRef="wrc_task_schedule_courier" targetRef="wrc_parallel_join" />
    <sequenceFlow id="wrc_sf_join_update" sourceRef="wrc_parallel_join" targetRef="wrc_task_update_inventory" />
    <sequenceFlow id="wrc_sf_update_close" sourceRef="wrc_task_update_inventory" targetRef="wrc_task_close_claim" />

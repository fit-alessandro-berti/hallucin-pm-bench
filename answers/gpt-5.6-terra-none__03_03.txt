<?xml version="1.0" encoding="UTF-8"?>
<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                  id="wrc_definitions"
                  targetNamespace="http://example.com/warranty-replacement">
  <bpmn:process id="wrc_process" name="Warranty-Replacement v1.3" isExecutable="false">

    <bpmn:startEvent id="wrc_startEvent" name="online warranty claim">
      <bpmn:outgoing>wrc_flow_start_validate</bpmn:outgoing>
    </bpmn:startEvent>

    <bpmn:task id="wrc_validateClaim" name="validates the claim">
      <bpmn:incoming>wrc_flow_start_validate</bpmn:incoming>
      <bpmn:incoming>wrc_flow_revalidate</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_validate_complete</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_completenessGateway" name="claim is complete?">
      <bpmn:incoming>wrc_flow_validate_complete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_incomplete</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_complete</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_requestMissingInformation" name="requests missing information">
      <bpmn:incoming>wrc_flow_incomplete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_request_wait</bpmn:outgoing>
    </bpmn:task>

    <bpmn:intermediateCatchEvent id="wrc_informationReceived" name="information is received">
      <bpmn:incoming>wrc_flow_request_wait</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_revalidate</bpmn:outgoing>
      <bpmn:messageEventDefinition id="wrc_messageDefinition_informationReceived"/>
    </bpmn:intermediateCatchEvent>

    <bpmn:task id="wrc_checkWarrantyEligibility" name="checks warranty eligibility">
      <bpmn:incoming>wrc_flow_complete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_eligibility_decision</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_warrantyGateway" name="within warranty?">
      <bpmn:incoming>wrc_flow_eligibility_decision</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_within_warranty</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_out_of_warranty</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_sendRejectionNotice" name="send rejection notice">
      <bpmn:incoming>wrc_flow_out_of_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_rejection_end</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_rejectionEndEvent" name="rejection">
      <bpmn:incoming>wrc_flow_rejection_end</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:parallelGateway id="wrc_parallelSplitGateway" name="parallel split">
      <bpmn:incoming>wrc_flow_within_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_create_shipment</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_schedule_pickup</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_createShipmentOrder" name="Create shipment order">
      <bpmn:incoming>wrc_flow_create_shipment</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_shipment_join</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_scheduleCourierPickup" name="Schedule courier pickup">
      <bpmn:incoming>wrc_flow_schedule_pickup</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_pickup_join</bpmn:outgoing>
    </bpmn:task>

    <bpmn:parallelGateway id="wrc_parallelJoinGateway" name="parallel join">
      <bpmn:incoming>wrc_flow_shipment_join</bpmn:incoming>
      <bpmn:incoming>wrc_flow_pickup_join</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_update_inventory</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_updateInventoryRecords" name="update inventory records">
      <bpmn:incoming>wrc_flow_update_inventory</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_close_claim</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_closeClaim" name="close the claim">
      <bpmn:incoming>wrc_flow_close_claim</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_success_end</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_successEndEvent" name="successful completion">
      <bpmn:incoming>wrc_flow_success_end</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:sequenceFlow id="wrc_flow_start_validate" sourceRef="wrc_startEvent" targetRef="wrc_validateClaim"/>
    <bpmn:sequenceFlow id="wrc_flow_validate_complete" sourceRef="wrc_validateClaim" targetRef="wrc_completenessGateway"/>
    <bpmn:sequenceFlow id="wrc_flow_incomplete" name="incomplete" sourceRef="wrc_completenessGateway" targetRef="wrc_requestMissingInformation"/>
    <bpmn:sequenceFlow id="wrc_flow_request_wait" sourceRef="wrc_requestMissingInformation" targetRef="wrc_informationReceived"/>
    <bpmn:sequenceFlow id="wrc_flow_revalidate" sourceRef="wrc_informationReceived" targetRef="wrc_validateClaim"/>
    <bpmn:sequenceFlow id="wrc_flow_complete" name="complete" sourceRef="wrc_completenessGateway" targetRef="wrc_checkWarrantyEligibility"/>
    <bpmn:sequenceFlow id="wrc_flow_eligibility_decision" sourceRef="wrc_checkWarrantyEligibility" targetRef="wrc_warrantyGateway"/>
    <bpmn:sequenceFlow id="wrc_flow_within_warranty" name="within warranty" sourceRef="wrc_warrantyGateway" targetRef="wrc_parallelSplitGateway"/>
    <bpmn:sequenceFlow id="wrc_flow_out_of_warranty" name="out of warranty" sourceRef="wrc_warrantyGateway" targetRef="wrc_sendRejectionNotice"/>
    <bpmn:sequenceFlow id="wrc_flow_rejection_end" sourceRef="wrc_sendRejectionNotice" targetRef="wrc_rejectionEndEvent"/>
    <bpmn:sequenceFlow id="wrc_flow_create_shipment" sourceRef="wrc_parallelSplitGateway" targetRef="wrc_createShipmentOrder"/>
    <bpmn:sequenceFlow id="wrc_flow_schedule_pickup" sourceRef="wrc_parallelSplitGateway" targetRef="wrc_scheduleCourierPickup"/>
    <bpmn:sequenceFlow id="wrc_flow_shipment_join" sourceRef="wrc_createShipmentOrder" targetRef="wrc_parallelJoinGateway"/>
    <bpmn:sequenceFlow id="wrc_flow_pickup_join" sourceRef="wrc_scheduleCourierPickup" targetRef="wrc_parallelJoinGateway"/>
    <bpmn:sequenceFlow id="wrc_flow_update_inventory" sourceRef="wrc_parallelJoinGateway" targetRef="wrc_updateInventoryRecords"/>
    <bpmn:sequenceFlow id="wrc_flow_close_claim" sourceRef="wrc_updateInventoryRecords" targetRef="wrc_closeClaim"/>
    <bpmn:sequenceFlow id="wrc_flow_success_end" sourceRef="wrc_closeClaim" targetRef="wrc_successEndEvent"/>

  </bpmn:process>
</bpmn:definitions>

      <conditionExpression xsi:type="tFormalExpression" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">

    </sequenceFlow>

    <sequenceFlow id="wrc_flow4" sourceRef="wrc_request_info" targetRef="wrc_validate_claim" />

      </conditionExpression>
    </sequenceFlow>
    <serviceTask id="wrc_check_warranty" name="Check warranty eligibility" />

    </sequenceFlow>

    <sequenceFlow id="wrc_flow9" sourceRef="wrc_warranty_decision_gateway" targetRef="wrc_parallel_split">

      <bpmn:outgoing>wrc_flow_validate_completeness</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_completeness_gateway" name="Claim completeness">
      <bpmn:incoming>wrc_flow_validate_completeness</bpmn:incoming>

      <bpmn:outgoing>wrc_flow_complete</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_request_missing_information" name="requests missing information">
      <bpmn:incoming>wrc_flow_incomplete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_request_wait</bpmn:outgoing>
    </bpmn:task>

    <bpmn:intermediateCatchEvent id="wrc_missing_information_received" name="missing information is received">
      <bpmn:incoming>wrc_flow_request_wait</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_received_revalidate</bpmn:outgoing>
      <bpmn:messageEventDefinition
          id="wrc_missing_information_message_definition"
          messageRef="wrc_message_missing_information"/>
    </bpmn:intermediateCatchEvent>

    <bpmn:task id="wrc_check_warranty_eligibility" name="checks warranty eligibility">
      <bpmn:incoming>wrc_flow_complete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_check_warranty</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_warranty_gateway" name="Warranty eligibility">
      <bpmn:incoming>wrc_flow_check_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_within_warranty</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_out_of_warranty</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_send_rejection_notice" name="send rejection notice">
      <bpmn:incoming>wrc_flow_out_of_warranty</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_rejection_end</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_rejection_end" name="Rejection">
      <bpmn:incoming>wrc_flow_rejection_end</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:parallelGateway id="wrc_parallel_split">

      <bpmn:outgoing>wrc_flow_create_shipment</bpmn:outgoing>

    <bpmn:endEvent id="wrc_success_end" name="Successful completion">

    <bpmn:sequenceFlow
        id="wrc_flow_check_warranty"

    <bpmn:sequenceFlow

    </bpmn:sequenceFlow>

        name="out of warranty"

    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow
        id="wrc_flow_rejection_end"
        sourceRef="wrc_send_rejection_notice"

        id="wrc_flow_create_shipment"

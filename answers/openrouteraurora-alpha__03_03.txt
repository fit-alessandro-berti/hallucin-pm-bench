```xml
<?xml version="1.0" encoding="UTF-8"?>
<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI"
                  xmlns:dc="http://www.omg.org/spec/DD/20100524/DC"
                  xmlns:di="http://www.omg.org/spec/DD/20100524/DI"
                  xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                  targetNamespace="http://example.com/warranty-replacement"
                  xsi:schemaLocation="http://www.omg.org/spec/BPMN/20100524/MODEL BPMN20.xsd">

  <bpmn:process id="wrc_WarrantyReplacement" name="Warranty-Replacement v1.3" isExecutable="true">

    <!-- Start -->
    <bpmn:startEvent id="wrc_start" name="Start"/>

    <!-- Step 2 – Validation -->
    <bpmn:task id="wrc_validates_the_claim" name="validates the claim"/>
    <bpmn:exclusiveGateway id="wrc_gateway_completeness" name="Completeness Check"/>
    <bpmn:task id="wrc_requests_missing_information" name="requests missing information"/>

    <!-- Step 3 – Warranty decision -->
    <bpmn:task id="wrc_checks_warranty_eligibility" name="checks warranty eligibility"/>
    <bpmn:exclusiveGateway id="wrc_gateway_warranty_decision" name="Warranty Decision"/>
    <bpmn:task id="wrc_send_rejection_notice" name="send rejection notice"/>
    <bpmn:endEvent id="wrc_end_rejection" name="Rejection End"/>

    <!-- Step 4 – Parallel activities -->
    <bpmn:parallelGateway id="wrc_parallel_split" name="Parallel Split"/>
    <bpmn:task id="wrc_Create_shipment_order" name="Create shipment order"/>
    <bpmn:task id="wrc_Schedule_courier_pickup" name="Schedule courier pickup"/>
    <bpmn:parallelGateway id="wrc_parallel_join" name="Parallel Join"/>

    <!-- Step 5 – Closing -->
    <bpmn:task id="wrc_update_inventory_records" name="update inventory records"/>
    <bpmn:task id="wrc_close_the_claim" name="close the claim"/>
    <bpmn:endEvent id="wrc_end_success" name="Success End"/>

    <!-- Sequence flows -->
    <bpmn:sequenceFlow id="wrc_flow_start_to_validates"
                      sourceRef="wrc_start"
                      targetRef="wrc_validates_the_claim"/>

    <bpmn:sequenceFlow id="wrc_flow_validates_to_gateway"
                      sourceRef="wrc_validates_the_claim"
                      targetRef="wrc_gateway_completeness"/>

    <!-- Incomplete → request missing info -->
    <bpmn:sequenceFlow id="wrc_flow_gateway_to_requests"
                      sourceRef="wrc_gateway_completeness"
                      targetRef="wrc_requests_missing_information">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression">incomplete</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <!-- Request missing info → re‑validate -->
    <bpmn:sequenceFlow id="wrc_flow_requests_to_validates"
                      sourceRef="wrc_requests_missing_information"
                      targetRef="wrc_validates_the_claim"/>

    <!-- Complete → eligibility check -->
    <bpmn:sequenceFlow id="wrc_flow_gateway_to_checks"
                      sourceRef="wrc_gateway_completeness"
                      targetRef="wrc_checks_warranty_eligibility">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression">complete</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <!-- Eligibility → warranty decision -->
    <bpmn:sequenceFlow id="wrc_flow_checks_to_warranty_gateway"
                      sourceRef="wrc_checks_warranty_eligibility"
                      targetRef="wrc_gateway_warranty_decision"/>

    <!-- Out of warranty → rejection -->
    <bpmn:sequenceFlow id="wrc_flow_warranty_gateway_to_reject"
                      sourceRef="wrc_gateway_warranty_decision"
                      targetRef="wrc_send_rejection_notice">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression">out of warranty</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow id="wrc_flow_reject_to_end"
                      sourceRef="wrc_send_rejection_notice"
                      targetRef="wrc_end_rejection"/>

    <!-- Within warranty → parallel split -->
    <bpmn:sequenceFlow id="wrc_flow_warranty_gateway_to_parallel"
                      sourceRef="wrc_gateway_warranty_decision"
                      targetRef="wrc_parallel_split">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression">within warranty</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <!-- Parallel activities -->
    <bpmn:sequenceFlow id="wrc_flow_parallel_to_shipment"
                      sourceRef="wrc_parallel_split"
                      targetRef="wrc_Create_shipment_order"/>
    <bpmn:sequenceFlow id="wrc_flow_parallel_to_courier"
                      sourceRef="wrc_parallel_split"
                      targetRef="wrc_Schedule_courier_pickup"/>

    <!-- Parallel join -->
    <bpmn:sequenceFlow id="wrc_flow_shipment_to_join"
                      sourceRef="wrc_Create_shipment_order"
                      targetRef="wrc_parallel_join"/>
    <bpmn:sequenceFlow id="wrc_flow_courier_to_join"
                      sourceRef="wrc_Schedule_courier_pickup"
                      targetRef="wrc_parallel_join"/>

    <!-- Closing steps -->
    <bpmn:sequenceFlow id="wrc_flow_join_to_update"
                      sourceRef="wrc_parallel_join"
                      targetRef="wrc_update_inventory_records"/>
    <bpmn:sequenceFlow id="wrc_flow_update_to_close"
                      sourceRef="wrc_update_inventory_records"
                      targetRef="wrc_close_the_claim"/>
    <bpmn:sequenceFlow id="wrc_flow_close_to_end"
                      sourceRef="wrc_close_the_claim"
                      targetRef="wrc_end_success"/>

  </bpmn:process>

</bpmn:definitions>
```

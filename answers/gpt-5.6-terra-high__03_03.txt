<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI"

                  id="wrc_Definitions"
                  targetNamespace="http://example.com/warranty-replacement">
  
  <bpmn:message id="wrc_Message_MissingInformationReceived" name="missing information is received"/>

  <bpmn:process id="wrc_WarrantyReplacementProcess" name="Warranty-Replacement v1.3" isExecutable="false">

    <bpmn:startEvent id="wrc_StartEvent_OnlineWarrantyClaim" name="customer submits an online warranty claim">
      <bpmn:outgoing>wrc_Flow_StartToValidate</bpmn:outgoing>
    </bpmn:startEvent>

    <bpmn:task id="wrc_Task_ValidateClaim" name="validates the claim">
      <bpmn:incoming>wrc_Flow_StartToValidate</bpmn:incoming>
      <bpmn:incoming>wrc_Flow_MissingInfoReceivedToValidate</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_ValidateToCompleteness</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_Gateway_Completeness" name="claim complete?" gatewayDirection="Diverging">
      <bpmn:incoming>wrc_Flow_ValidateToCompleteness</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_ClaimIncomplete</bpmn:outgoing>
      <bpmn:outgoing>wrc_Flow_ClaimComplete</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_Task_RequestMissingInformation" name="requests missing information">
      <bpmn:incoming>wrc_Flow_ClaimIncomplete</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_RequestToMissingInfoReceived</bpmn:outgoing>
    </bpmn:task>

    <bpmn:intermediateCatchEvent id="wrc_Event_MissingInformationReceived" name="missing information is received">
      <bpmn:incoming>wrc_Flow_RequestToMissingInfoReceived</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_MissingInfoReceivedToValidate</bpmn:outgoing>
      <bpmn:messageEventDefinition id="wrc_MessageEventDefinition_MissingInformationReceived"
                                   messageRef="wrc_Message_MissingInformationReceived"/>
    </bpmn:intermediateCatchEvent>

    <bpmn:task id="wrc_Task_CheckWarrantyEligibility" name="checks warranty eligibility">
      <bpmn:incoming>wrc_Flow_ClaimComplete</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_EligibilityToWarrantyDecision</bpmn:outgoing>
    </bpmn:task>

    <bpmn:exclusiveGateway id="wrc_Gateway_WarrantyDecision" name="within warranty?" gatewayDirection="Diverging">
      <bpmn:incoming>wrc_Flow_EligibilityToWarrantyDecision</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_OutOfWarranty</bpmn:outgoing>
      <bpmn:outgoing>wrc_Flow_WithinWarranty</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <bpmn:task id="wrc_Task_SendRejectionNotice" name="send rejection notice">
      <bpmn:incoming>wrc_Flow_OutOfWarranty</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_RejectionToEnd</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_EndEvent_Rejection" name="rejection">
      <bpmn:incoming>wrc_Flow_RejectionToEnd</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:parallelGateway id="wrc_Gateway_ParallelSplit" gatewayDirection="Diverging">
      <bpmn:incoming>wrc_Flow_WithinWarranty</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_SplitToShipmentOrder</bpmn:outgoing>
      <bpmn:outgoing>wrc_Flow_SplitToCourierPickup</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_Task_CreateShipmentOrder" name="Create shipment order">
      <bpmn:incoming>wrc_Flow_SplitToShipmentOrder</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_ShipmentOrderToJoin</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_Task_ScheduleCourierPickup" name="Schedule courier pickup">
      <bpmn:incoming>wrc_Flow_SplitToCourierPickup</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_CourierPickupToJoin</bpmn:outgoing>
    </bpmn:task>

    <bpmn:parallelGateway id="wrc_Gateway_ParallelJoin" gatewayDirection="Converging">
      <bpmn:incoming>wrc_Flow_ShipmentOrderToJoin</bpmn:incoming>
      <bpmn:incoming>wrc_Flow_CourierPickupToJoin</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_JoinToUpdateInventory</bpmn:outgoing>
    </bpmn:parallelGateway>

    <bpmn:task id="wrc_Task_UpdateInventoryRecords" name="update inventory records">
      <bpmn:incoming>wrc_Flow_JoinToUpdateInventory</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_UpdateInventoryToCloseClaim</bpmn:outgoing>
    </bpmn:task>

    <bpmn:task id="wrc_Task_CloseClaim" name="close the claim">
      <bpmn:incoming>wrc_Flow_UpdateInventoryToCloseClaim</bpmn:incoming>
      <bpmn:outgoing>wrc_Flow_CloseClaimToSuccessEnd</bpmn:outgoing>
    </bpmn:task>

    <bpmn:endEvent id="wrc_EndEvent_SuccessfulCompletion" name="successful completion">
      <bpmn:incoming>wrc_Flow_CloseClaimToSuccessEnd</bpmn:incoming>
    </bpmn:endEvent>

    <bpmn:sequenceFlow id="wrc_Flow_StartToValidate"
                       sourceRef="wrc_StartEvent_OnlineWarrantyClaim"
                       targetRef="wrc_Task_ValidateClaim"/>

    <bpmn:sequenceFlow id="wrc_Flow_ValidateToCompleteness"
                       sourceRef="wrc_Task_ValidateClaim"
                       targetRef="wrc_Gateway_Completeness"/>

    <bpmn:sequenceFlow id="wrc_Flow_ClaimIncomplete"
                       name="incomplete"
                       sourceRef="wrc_Gateway_Completeness"
                       targetRef="wrc_Task_RequestMissingInformation">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression"><![CDATA[claim is incomplete]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow id="wrc_Flow_RequestToMissingInfoReceived"
                       sourceRef="wrc_Task_RequestMissingInformation"
                       targetRef="wrc_Event_MissingInformationReceived"/>

    <bpmn:sequenceFlow id="wrc_Flow_MissingInfoReceivedToValidate"
                       sourceRef="wrc_Event_MissingInformationReceived"
                       targetRef="wrc_Task_ValidateClaim"/>

    <bpmn:sequenceFlow id="wrc_Flow_ClaimComplete"
                       name="complete"
                       sourceRef="wrc_Gateway_Completeness"
                       targetRef="wrc_Task_CheckWarrantyEligibility">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression"><![CDATA[claim is complete]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow id="wrc_Flow_EligibilityToWarrantyDecision"
                       sourceRef="wrc_Task_CheckWarrantyEligibility"
                       targetRef="wrc_Gateway_WarrantyDecision"/>

    <bpmn:sequenceFlow id="wrc_Flow_OutOfWarranty"
                       name="out of warranty"
                       sourceRef="wrc_Gateway_WarrantyDecision"
                       targetRef="wrc_Task_SendRejectionNotice">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression"><![CDATA[product is out of warranty]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow id="wrc_Flow_RejectionToEnd"
                       sourceRef="wrc_Task_SendRejectionNotice"
                       targetRef="wrc_EndEvent_Rejection"/>

    <bpmn:sequenceFlow id="wrc_Flow_WithinWarranty"
                       name="within warranty"
                       sourceRef="wrc_Gateway_WarrantyDecision"
                       targetRef="wrc_Gateway_ParallelSplit">
      <bpmn:conditionExpression xsi:type="bpmn:tFormalExpression"><![CDATA[product is within warranty]]></bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:sequenceFlow id="wrc_Flow_SplitToShipmentOrder"
                       sourceRef="wrc_Gateway_ParallelSplit"
                       targetRef="wrc_Task_CreateShipmentOrder"/>

    <bpmn:sequenceFlow id="wrc_Flow_SplitToCourierPickup"
                       sourceRef="wrc_Gateway_ParallelSplit"
                       targetRef="wrc_Task_ScheduleCourierPickup"/>

    <bpmn:sequenceFlow id="wrc_Flow_ShipmentOrderToJoin"
                       sourceRef="wrc_Task_CreateShipmentOrder"
                       targetRef="wrc_Gateway_ParallelJoin"/>

    <bpmn:sequenceFlow id="wrc_Flow_CourierPickupToJoin"
                       sourceRef="wrc_Task_ScheduleCourierPickup"
                       targetRef="wrc_Gateway_ParallelJoin"/>

    <bpmn:sequenceFlow id="wrc_Flow_JoinToUpdateInventory"
                       sourceRef="wrc_Gateway_ParallelJoin"
                       targetRef="wrc_Task_UpdateInventoryRecords"/>

    <bpmn:sequenceFlow id="wrc_Flow_UpdateInventoryToCloseClaim"
                       sourceRef="wrc_Task_UpdateInventoryRecords"
                       targetRef="wrc_Task_CloseClaim"/>

    <bpmn:sequenceFlow id="wrc_Flow_CloseClaimToSuccessEnd"
                       sourceRef="wrc_Task_CloseClaim"
                       targetRef="wrc_EndEvent_SuccessfulCompletion"/>

  </bpmn:process>

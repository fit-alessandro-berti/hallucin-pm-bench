    </sequenceFlow>

      <conditionExpression xsi:type="tFormalExpression">Product is out of warranty</conditionExpression>
    </sequenceFlow>

    <sequenceFlow id="wrc_flow_send_rejection_to_rejection_end"

    <sequenceFlow id="wrc_flow_parallel_split_to_schedule_pickup"
                  sourceRef="wrc_parallel_split_gateway"

    <sequenceFlow id="wrc_flow_close_claim_to_success_end"
                  sourceRef="wrc_close_claim_task"

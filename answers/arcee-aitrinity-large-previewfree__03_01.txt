```xml
<definitions xmlns="http://www.omg.org/spec/BPMN/20100524/MODEL"
             xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
             targetNamespace="http://acmetech.com/support">
  <process id="SupportTicketProcess" isExecutable="false">
    <startEvent id="start"/>
    <sequenceFlow id="flow1" sourceRef="start" targetRef="classify"/>
    <task id="classify" name="Classify Ticket"/>
    <exclusiveGateway id="gw1" name="Password Reset?"/>
    <sequenceFlow id="flow2" sourceRef="classify" targetRef="gw1"/>
    <sequenceFlow id="flow3" sourceRef="gw1" targetRef="reset">
      <conditionExpression xsi:type="tFormalExpression">return classification == "Password Reset";</conditionExpression>
    </sequenceFlow>
    <task id="reset" name="Provide Reset Instructions"/>
    <sequenceFlow id="flow4" sourceRef="gw1" targetRef="diagnosis">
      <conditionExpression xsi:type="tFormalExpression">return classification != "Password Reset";</conditionExpression>
    </sequenceFlow>
    <task id="diagnosis" name="Initial Diagnosis"/>
    <exclusiveGateway id="gw2" name="Resolve or Forward?"/>
    <sequenceFlow id="flow5" sourceRef="diagnosis" targetRef="gw2"/>
    <sequenceFlow id="flow6" sourceRef="gw2" targetRef="resolve">
      <conditionExpression xsi:type="tFormalExpression">return resolveDirectly;</conditionExpression>
    </sequenceFlow>
    <task id="resolve" name="Resolve Issue"/>
    <sequenceFlow id="flow7" sourceRef="gw2" targetRef="forward">
      <conditionExpression xsi:type="tFormalExpression">return forwardToSpecialist;</conditionExpression>
    </sequenceFlow>
    <task id="forward" name="Forward to Specialist"/>
    <sequenceFlow id="flow8" sourceRef="forward" targetRef="resolve"/>
    <task id="confirm" name="Confirm with Customer"/>
    <exclusiveGateway id="gw3" name="Customer Confirmed?"/>
    <sequenceFlow id="flow9" sourceRef="resolve" targetRef="confirm"/>
    <sequenceFlow id="flow10" sourceRef="confirm" targetRef="gw3"/>
    <sequenceFlow id="flow11" sourceRef="gw3" targetRef="end">
      <conditionExpression xsi:type="tFormalExpression">return confirmed;</conditionExpression>
    </sequenceFlow>
    <sequenceFlow id="flow12" sourceRef="gw3" targetRef="diagnosis">
      <conditionExpression xsi:type="tFormalExpression">return !confirmed;</conditionExpression>
    </sequenceFlow>
    <endEvent id="end"/>
  </process>
</definitions>
```

# Functions to help with comparison
def parse_bpmn(xml_str):
    return fromstring(xml_str).findall('.//process')[0]

def extract_tasks(process):
    return {task.attrib['id']: task.attrib['name'] for task in process.findall('.//task')}

def extract_gateways(process):
    gateways = {}
    for gateway in process.findall('.//gateway'):
        gateways[gateway.attrib['id']] = (gateway.tag, gateway.attrib['name'])
    return gateways

def compare_tasks(v1_tasks, v2_tasks):
    added = {id: name for id, name in v2_tasks.items() if id not in v1_tasks}
    removed = {id: name for id, name in v1_tasks.items() if id not in v2_tasks}
    return added, removed

def compare_gateways(v1_gateways, v2_gateways):
    changes = [{id: (tag1, name1, tag2)} for id, ((tag1, name1), (tag2, _)) in 
               v2_gateways.items() if (tag1, name1) != (tag2, name2)]
    if not changes:
        return None
    return changes

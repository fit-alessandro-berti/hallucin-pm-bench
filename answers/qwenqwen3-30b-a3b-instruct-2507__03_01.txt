<definitions xmlns="http://www.omg.org/spec/BPMN/20100524/MODEL"
             xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI"
             id="AcmeTechSupportProcess"
             targetNamespace="http://acmetech.com/bpmn/support">
  <process id="SupportProcess" isExecutable="true">
    <startEvent id="start" />
    <sequenceFlow id="sf1" sourceRef="start" targetRef="classify" />

    <serviceTask id="classify" name="Classify Ticket" />

    <sequenceFlow id="sf2" sourceRef="classify" targetRef="gw1" />

    <exclusiveGateway id="gw1" name="Is Password Reset?" />

    <sequenceFlow id="sf3" sourceRef="gw1" targetRef="provideReset" conditionExpression="=${classification == 'Password Reset'}" />
    <sequenceFlow id="sf4" sourceRef="gw1" targetRef="diagnose" conditionExpression="=${classification != 'Password Reset'}" />

    <serviceTask id="provideReset" name="Provide Reset Instructions" />
    <sequenceFlow id="sf5" sourceRef="provideReset" targetRef="confirm" />

    <serviceTask id="diagnose" name="Initial Diagnosis" />
    <sequenceFlow id="sf6" sourceRef="diagnose" targetRef="gw2" />

    <exclusiveGateway id="gw2" name="Can Issue Be Resolved?" />

    <sequenceFlow id="sf7" sourceRef="gw2" targetRef="resolve" conditionExpression="=${resolution == 'Yes'}" />
    <sequenceFlow id="sf8" sourceRef="gw2" targetRef="forward" conditionExpression="=${resolution == 'No'}" />

    <serviceTask id="resolve" name="Resolve Issue" />
    <sequenceFlow id="sf9" sourceRef="resolve" targetRef="confirm" />

    <serviceTask id="forward" name="Forward to Specialist" />
    <sequenceFlow id="sf10" sourceRef="forward" targetRef="specialistResolve" />

    <serviceTask id="specialistResolve" name="Resolve Issue" />
    <sequenceFlow id="sf11" sourceRef="specialistResolve" targetRef="confirm" />

    <serviceTask id="confirm" name="Confirm with Customer" />
    <sequenceFlow id="sf12" sourceRef="confirm" targetRef="gw3" />

    <exclusiveGateway id="gw3" name="Customer Confirmed?" />

    <sequenceFlow id="sf13" sourceRef="gw3" targetRef="close" conditionExpression="=${confirmation == 'Yes'}" />
    <sequenceFlow id="sf14" sourceRef="gw3" targetRef="diagnose" conditionExpression="=${confirmation == 'No'}" />

    <endEvent id="close" />
    <sequenceFlow id="sf15" sourceRef="close" targetRef="end" />

    <endEvent id="end" />
  </process>

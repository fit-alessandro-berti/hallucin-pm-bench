```xml
<?xml version="1.0" encoding="UTF-8"?>
<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI"
                  xmlns:dc="http://www.omg.org/spec/DD/20100524/DC"
                  xmlns:di="http://www.omg.org/spec/DD/20100524/DI"
                  xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                  id="wrc_definitions"
                  targetNamespace="http://warranty-replacement/bpmn">

  <bpmn:process id="wrc_process" name="Warranty-Replacement v1.3" isExecutable="false">

    <!-- Start Event -->
    <bpmn:startEvent id="wrc_start" name="Customer submits online warranty claim">
      <bpmn:outgoing>wrc_flow_start_to_validate</bpmn:outgoing>
    </bpmn:startEvent>

    <!-- Task: validates the claim -->
    <bpmn:task id="wrc_validate_claim" name="validates the claim">
      <bpmn:incoming>wrc_flow_start_to_validate</bpmn:incoming>
      <bpmn:incoming>wrc_flow_request_to_validate</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_validate_to_complete</bpmn:outgoing>
    </bpmn:task>

    <!-- Exclusive Gateway: claim completeness -->
    <bpmn:exclusiveGateway id="wrc_claim_complete" name="Claim complete?">
      <bpmn:incoming>wrc_flow_validate_to_complete</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_complete_no</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_complete_yes</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <!-- Task: requests missing information -->
    <bpmn:task id="wrc_request_info" name="requests missing information">
      <bpmn:incoming>wrc_flow_complete_no</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_request_to_validate</bpmn:outgoing>
    </bpmn:task>

    <!-- Task: checks warranty eligibility -->
    <bpmn:task id="wrc_check_eligibility" name="checks warranty eligibility">
      <bpmn:incoming>wrc_flow_complete_yes</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_eligibility_to_decision</bpmn:outgoing>
    </bpmn:task>

    <!-- Exclusive Gateway: warranty decision -->
    <bpmn:exclusiveGateway id="wrc_warranty_check" name="Within warranty?">
      <bpmn:incoming>wrc_flow_eligibility_to_decision</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_warranty_out</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_warranty_in</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <!-- Task: send rejection notice -->
    <bpmn:task id="wrc_send_rejection" name="send rejection notice">
      <bpmn:incoming>wrc_flow_warranty_out</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_rejection_to_end</bpmn:outgoing>
    </bpmn:task>

    <!-- End Event: rejection -->
    <bpmn:endEvent id="wrc_end_rejection" name="Rejection End">
      <bpmn:incoming>wrc_flow_rejection_to_end</bpmn:incoming>
    </bpmn:endEvent>

    <!-- Parallel Split Gateway -->
    <bpmn:parallelGateway id="wrc_parallel_split" name="Split parallel activities">
      <bpmn:incoming>wrc_flow_warranty_in</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_split_to_shipment</bpmn:outgoing>
      <bpmn:outgoing>wrc_flow_split_to_pickup</bpmn:outgoing>
    </bpmn:parallelGateway>

    <!-- Task: Create shipment order -->
    <bpmn:task id="wrc_create_shipment" name="Create shipment order">
      <bpmn:incoming>wrc_flow_split_to_shipment</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_shipment_to_join</bpmn:outgoing>
    </bpmn:task>

    <!-- Task: Schedule courier pickup -->
    <bpmn:task id="wrc_schedule_pickup" name="Schedule courier pickup">
      <bpmn:incoming>wrc_flow_split_to_pickup</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_pickup_to_join</bpmn:outgoing>
    </bpmn:task>

    <!-- Parallel Join Gateway -->
    <bpmn:parallelGateway id="wrc_parallel_join" name="Join parallel activities">
      <bpmn:incoming>wrc_flow_shipment_to_join</bpmn:incoming>
      <bpmn:incoming>wrc_flow_pickup_to_join</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_join_to_inventory</bpmn:outgoing>
    </bpmn:parallelGateway>

    <!-- Task: update inventory records -->
    <bpmn:task id="wrc_update_inventory" name="update inventory records">
      <bpmn:incoming>wrc_flow_join_to_inventory</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_inventory_to_close</bpmn:outgoing>
    </bpmn:task>

    <!-- Task: close the claim -->
    <bpmn:task id="wrc_close_claim" name="close the claim">
      <bpmn:incoming>wrc_flow_inventory_to_close</bpmn:incoming>
      <bpmn:outgoing>wrc_flow_close_to_end</bpmn:outgoing>
    </bpmn:task>

    <!-- End Event: successful completion -->
    <bpmn:endEvent id="wrc_end_success" name="Success End">
      <bpmn:incoming>wrc_flow_close_to_end</bpmn:incoming>
    </bpmn:endEvent>

    <!-- Sequence Flows -->
    <bpmn:sequenceFlow id="wrc_flow_start_to_validate"     sourceRef="wrc_start"              targetRef="wrc_validate_claim"/>
    <bpmn:sequenceFlow id="wrc_flow_validate_to_complete"  sourceRef="wrc_validate_claim"     targetRef="wrc_claim_complete"/>
    <bpmn:sequenceFlow id="wrc_flow_complete_no"  name="No"  sourceRef="wrc_claim_complete"   targetRef="wrc_request_info"/>
    <bpmn:sequenceFlow id="wrc_flow_complete_yes" name="Yes" sourceRef="wrc_claim_complete"   targetRef="wrc_check_eligibility"/>
    <bpmn:sequenceFlow id="wrc_flow_request_to_validate"   sourceRef="wrc_request_info"       targetRef="wrc_validate_claim"/>
    <bpmn:sequenceFlow id="wrc_flow_eligibility_to_decision" sourceRef="wrc_check_eligibility" targetRef="wrc_warranty_check"/>
    <bpmn:sequenceFlow id="wrc_flow_warranty_out"  name="Out of warranty" sourceRef="wrc_warranty_check" targetRef="wrc_send_rejection"/>
    <bpmn:sequenceFlow id="wrc_flow_warranty_in"   name="Within warranty" sourceRef="wrc_warranty_check" targetRef="wrc_parallel_split"/>
    <bpmn:sequenceFlow id="wrc_flow_rejection_to_end"      sourceRef="wrc_send_rejection"     targetRef="wrc_end_rejection"/>
    <bpmn:sequenceFlow id="wrc_flow_split_to_shipment"     sourceRef="wrc_parallel_split"     targetRef="wrc_create_shipment"/>
    <bpmn:sequenceFlow id="wrc_flow_split_to_pickup"       sourceRef="wrc_parallel_split"     targetRef="wrc_schedule_pickup"/>
    <bpmn:sequenceFlow id="wrc_flow_shipment_to_join"      sourceRef="wrc_create_shipment"    targetRef="wrc_parallel_join"/>
    <bpmn:sequenceFlow id="wrc_flow_pickup_to_join"        sourceRef="wrc_schedule_pickup"    targetRef="wrc_parallel_join"/>
    <bpmn:sequenceFlow id="wrc_flow_join_to_inventory"     sourceRef="wrc_parallel_join"      targetRef="wrc_update_inventory"/>
    <bpmn:sequenceFlow id="wrc_flow_inventory_to_close"    sourceRef="wrc_update_inventory"   targetRef="wrc_close_claim"/>
    <bpmn:sequenceFlow id="wrc_flow_close_to_end"          sourceRef="wrc_close_claim"        targetRef="wrc_end_success"/>

  </bpmn:process>
</bpmn:definitions>
```
